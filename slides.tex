{\LARGE \bf Estratégias de Commitment}

\begin{description}
\item [Blind commitment:] Agente mantém suas intenções até acreditar que ela foi alcançada. Também é conhecido como {\it fanatical commitment}
\item [Single-minded commitment:] Agente mantém sua intenção até acreditar que ou sua intenção foi alcançada, ou que não é mais possível alcança-la.
\item [Open-minded commitment:] Agente mantém sua intenção até acreditar que ela possa ser alcançada.
\end{description}

}

\begin{verbatim}

      alpha=cabeça(pi)
      executa(alpha)
      pi=cauda(pi)
      pega percepção p
      B=revisa(B,p)
      if not soa_possivel(pi,I,B) then
         pi=planeja(B,I)
      end if
   end while
end while
\end{verbatim}

Agente reativo: quando um plano aparenta ter falhado, planeja novamente.

\clearpage
{\Large
{\LARGE \bf BDI - Believe, Desire, Intention}
\begin{description}
\item[Bel $i$ $\alpha$:] Agente $i$ acredita $\alpha$
\item[Des $i$ $\alpha$:] Agente $i$ deseja $\alpha$
\item[Int $i$ $\alpha$:] Agente $i$ intenciona $\alpha$
\end{description}
Algumas características:
\begin{itemize}
\item Permite quantificadores sobre os agentes
\item {\bf NÃO} permite quantificadores sobre as fórmulas
\end{itemize}
}

\clearpage
{\Large
{\LARGE \bf Temporal}
\begin{description}
\item[$\circ \alpha$:] $\alpha$ é verdade no próximo instante
\item[$\Diamond \alpha$:] $\alpha$ será verdadeiro em algum momento no futuro
\item[$\Box \alpha$:] $\alpha$ sempre é verdade
\item[$\alpha$ U $\beta$:] $\alpha$ é verdadeiro até que $\beta$ seja verdadeiro
\item[$\alpha$ W $\beta$:] $\alpha$ é verdadeiro a menos que $\beta$ seja verdadeiro
\end{description}
Quantificadores sobre caminhos
\begin{description}
\item[A$\alpha$:] $\alpha$ é verdadeiro em todos os caminhos
\item[E$\alpha$:] $\alpha$ é verdadeiro em algum caminho
\end{description}
}

\clearpage
{\Large
{\LARGE \bf Ações}
\begin{description}
\item[Happens $\sigma$:] A ação $\sigma$ acontece no próximo instante
\item[Achvs $\sigma$ $\alpha$:] Ocorre a ação $\sigma$ atingindo $\alpha$
\item[Agts $\sigma$ $g$:] Um grupo de agentes $g$ é requerido para fazer a ação $\sigma$
\end{description}
Construtores para ações
\begin{description}
\item[$\sigma ; \sigma '$:] $\sigma$ seguido por $\sigma '$
\item[$\sigma | \sigma '$:] $\sigma$ ou $\sigma '$
\item[$\sigma *$:] $\sigma$ repetido mais de uma vez
\item[$\alpha ?$:] $\alpha$ é satisfeito?
\end{description}
}
